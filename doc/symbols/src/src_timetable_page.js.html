1 /**
  2  * Parameters for the timetable page
  3  * 
  4  * http://usejsdoc.org/
  5  */
  6 
  7 /**
  8  * Title for the page, to be displayed in both TITLE and H1 tags.
  9  */
 10 const TITLE = "You Need A Timetable";
 11 
 12 /**
 13  * Number of hours in a week
 14  */
 15 const HOURS_IN_WEEK = 7 * 24;
 16 
 17 /**
 18  * View to be used when rendering timetable page
 19  */
 20 const VIEW = 'timetable_page';
 21 
 22 /**
 23  * Render the timetable page
 24  * 
 25  * @param res {ServerResponse} - Stream to send the rendered page to
 26  * @param timetable {Object} - Timetable to be rendered.  Each category should be a key of timetable, with the number of hours as the associated value
 27  */
 28 function render(res, timetable) {
 29 	res.render(VIEW, {
 30 		title: TITLE,
 31 		unbudgeted: HOURS_IN_WEEK,
 32 		timetable: timetable.getTimetable()
 33 	});
 34 }
 35 
 36 module.exports = { TITLE, HOURS_IN_WEEK, VIEW, render };
