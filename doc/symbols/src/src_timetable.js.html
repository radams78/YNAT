  9 /**
 10  * A timetable.  The Model class for the app.
 11  */
 12 class Timetable {
 13 	/**
 14 	 * Creates a timetable with the given initial data.
 15 	 * 
 16 	 * @param initial_data {Object} - initial contents of the timetable.  Each category is a key in initial_data,
 17 	 *    with the initial number of hours as its value.  May be omitted for an empty timetable.
 18 	 */
 19 	constructor (initial_data) {
 20 		this.timetable = (initial_data ? initial_data : {}); 
 21 	}
 22 	
 23 	/**
 24 	 * Add a category, or change the value of a category.
 25 	 * 
 26 	 * @param category {String} - Category to be added or changed
 27 	 * @param time {Number} - New number of hours
 28 	 */
 29 	setCategory(category, time) {
 30 		this.timetable[category] = time;
 31 	}
 32 	
 33 	/**
 34 	 * Returns the contents of the timetable.
 35 	 * 
 36 	 * @returns Contents of timetable as an object.  Each category is a key in this object, with the number of hours as its value.
 37 	 */
 38 	getTimetable() {
 39 		return this.timetable;
 40 	}
 41 	
 42 	/**
 43 	 * Add or modify several categories.
 44 	 * 
 45 	 * @param newdata {Object} - Categories to be added or modified.  Each category is a key in newdata, with the new number of hours as its value.
 46 	 */
 47 	addData(newdata) {
 48 		for (let category in newdata) {
 49 			this.setCategory(category, newdata[category]);
 50 		}
 51 	}
 52 	
 53 	/**
 54 	 * Return a JSON representation of the timetable.
 55 	 * 
 56 	 * @returns JSON representation of the timetable
 57 	 * @type String
 58 	 */
 59 	toJSON()  {
 60 		return JSON.stringify(this.timetable);
 61 	}
 62 }
 63 
 64 module.exports = { Timetable };
